### while 반복문
형식 : while(조건식) {반복실행문}
while(조건식)에서 조건식이 true일 때, {}내의 실행문이 반복적으로 이루어짐
주의할 점 : while 내부의 조건식이 특정 시점에 false가 되도록 미리 설계해둘 필요가 있음
false가 되는 시점을 지정해 주지 않는다면 계속 true 상태가 유지되서 무한 루프가 일어남
while문 내부에 if문 쓸 수 있음

```java
public static void main(String[] args) {
	// 1부터 100까지 출력하는 예시
	int i = 1;
	while (i < 101) {            			 	 
        System.out.println(i);
        i++;            // i를 증가시켜서 특정 시점에 조건이 false가 되도록 지정
    }
   
    // 1부터 100까지의 합을 구하는 예시
    int num1 = 1;
    int sum = 0;

    while(num1 < 101) {
		sum = sum + num1;
		num1++;
    }
    System.out.println(sum);
   
	// 1부터 100까지 짝수의 합만 구하는 예시
	int j = 0;
	int sum3 = 0;

	while (j < 101) {
		if (j % 2 == 0) {
			sum3 += j;
		}
		j++;
	}
	System.out.println(sum3);
    
    /*
    	1 2 3 ... 10
        11 12 13 ...20
        91 92 93...100 
        으로 출력되게 하는 예시
    */
    
    int i = 0;

	// 반복문을 100번 실행하는 예시
	while (i < 100) {
		++i;
		System.out.print(i + " ");

		if(i % 10 == 0) {
			System.out.println();
		}
	}

	// 반복문을 10번 실행하는 예시
	int  j = 1;

	while (j < 101) {
		System.out.println(j + " " + (j+1) + " " + (j+2) + " " + (j+3) + " " +(j+4)+ " " 
        +(j+5) + " " +(j+6) + " " +(j+7) + " " +(j+8) + " " +(j+9));
		j+=10;
	}
}
```
### 중첩 while문 (Nested - while)
```java
public static void main(String[] args) {
	// 중첩 while 문을 활용하여 구구단 출력 예시
	int dan = 2;

	while (dan < 10) {
		int num = 1;	// 반복실행문 1-a에 해당. 변수의 선언 및 초기화
		while (num < 10) {
			int result = dan * num;	
			System.out.println(dan + " x " + num + " = " + result);	// 반복실행문2
			num++;
		}
		dan++;	// 반복 실행문 1-b
	}
    
    // while 문 안에 선언된 num 변수는 while문(반복문) 외부에서 쓰지 못함
    // scope(범위)라는 것으로, 전역(전체영역) / 지역으로 나뉜다는 점이 중요
}
```

### for 반복문(for loop)
대부분의 경우 명확한 횟수가 고정되어 있을 때 사용하는 반복문
(while 문에는 반복 횟수를 고정시킨 상태로 시작했지만 for 문을 활용하는 것이 더 편리함)

```java
import java.util.Scanner;

public static void main(String[] args) {
	// 1부터 100까지 출력하는 for문
    for (int i = 1; i < 101; i++) {
    	System.out.println(i);
    }
    
    // 1부터 100까지 중 홀수 합을 구하는 for문 예시
    // 방법 1 : for문 내부에 if 문을 중첩적으로 써서 구하는 방식
    int sum = 0;
    for (int i = 0; i < 101; i++) {
    	if(i % 2 != 0) {
        	sum += i;
        }
    }
    System.out.println(sum);
    
    // 방법 2 : 증감값을 건드려서 구하는 방식
    int sum2 = 0;
    for ( int i = 1; i < 101; i+=2) {
    	sum2 += i;
    }
    System.out.println(sum2);

    // scanner로 숫자(n)을 입력 받아 짝수와 홀수의 합을 도출하는 예시
    Scanner scanner = new Scanner(System.in);
    System.out.print("1부터 몇까지 더하시겠습니까? >>> ");
    int num = scanner.nextInt();

    int sum = 0;
    int sumEven = 0;
    int sumOdd = 0;

    for (int i = 0; i < (num+1); i++) {
        sum += i;

        if(i % 2 == 0) {
            sumEven += i;
        }
        else {
            sumOdd += i;
        }
    }
    System.out.println("1부터 " + num + "까지 짝수의 합은 " + sumEven);
    System.out.println("1부터 " + num + "까지 홀수의 합은 " + sumOdd);
    
}
```

### - 중첩 for 문을 활용한 별찍기 예시
```java
public static void main(String[] args) {
    // 왼쪽으로 치우친 증가하는 별 찍기
    for(int i = 0; i < 6; i++) {
        for(int j = 0; j < i; j++) {
            System.out.print("*");
        }
        System.out.println();
    }
    
    // 오른쪽으로 치우친 감소하는 별 찍기
    // 방법 1
    for(int i = 0; i < 6; i++) {
        for(int j = 5; j - i > 0; j--){
            System.out.print("*");
        }
        System.out.println();
    }
    
    // 방법 2
    for(int i = 5; i > 0; i--) {
        for(int j = 0; i - j > 0; j++){
            System.out.print("*");
        }
        System.out.println();
    }
    
    // 오른쪽으로 치우친 증가하는 별 찍기
    for (int i = 0; i < 6; i++) {
        // 공백 for문
        for (int j = 0; j < 5 - i; j++) {
            System.out.print(" ");
        }

        //별 for문
        for (int k = 0; k < i; k++) {
            System.out.print("*");
        }
        System.out.println();
    }
    
    // 오른쪽으로 치우친 감소하는 별 찍기
    for (int i = 0; i < 6; i++) {
        // 공백을 책임지는 for문
        for (int j = 0; j < i; j++) {
            System.out.print(" ");
        }

        //별을 책임지는 for문
        for (int k = 0; k < 5 - i; k++) {
            System.out.print("*");
        }
        System.out.println();
    }
    
}
```
